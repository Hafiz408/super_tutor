```typescript
// ADVISE-03: Persist quiz score to backend SQLite when quiz reviewing starts.
// Non-fatal — adaptive persistence failure does not break the quiz flow.
useEffect(() => {
  if (quizPhase !== "reviewing" || !session || !sessionId) return;
  fetch(`${process.env.NEXT_PUBLIC_API_URL}/tutor/${sessionId}/adaptive`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      quiz_score: { correct: correctCount, total: session.quiz.length },
      focus_areas: [],
    }),
  })
    .then((res) => {
      if (res.ok) return res.json();
    })
    .then((data) => {
      if (data?.focus_areas !== undefined) {
        setTutorAdaptiveData((prev) => ({
          ...prev,
          quiz_score: { correct: correctCount, total: session.quiz.length },
          focus_areas: data.focus_areas,
        }));
      }
    })
    .catch(() => {
      // Non-fatal: adaptive persistence failure does not interrupt the quiz
    });
}, [quizPhase, sessionId]);
// Note: correctCount and session are intentionally excluded from deps to avoid
// repeated fires — quizPhase transition to "reviewing" is the single trigger.
// eslint-disable-next-line react-hooks/exhaustive-deps
```
