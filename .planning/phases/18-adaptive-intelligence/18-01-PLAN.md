```python
@router.post("/{session_id}/adaptive")
async def persist_adaptive_data(
    session_id: str,
    body: TutorAdaptiveRequest,
    traces_db: SqliteDb = Depends(get_traces_db),
):
    """
    Persist quiz score and Advisor-identified focus areas to the workflow session_state.
    Merges with existing adaptive data — never overwrites; focus_areas are deduplicated.
    Uses build_session_workflow to write to the workflow row (NOT the tutor Team row).
    See RESEARCH.md Pitfall 3: workflow row uses session_id, tutor Team row uses tutor:{session_id}:{reset_id}.
    """
    wf = build_session_workflow(session_id=session_id, session_db=traces_db)
    existing = wf.get_session(session_id=session_id)
    if existing is None:
        raise HTTPException(status_code=404, detail=f"Session '{session_id}' not found.")

    state = (existing.session_data or {}).get("session_state", {})
    existing_data = state.get("tutor_adaptive_data", {})
    existing_areas = existing_data.get("focus_areas", [])
    new_areas = sorted(set(existing_areas) | set(body.focus_areas))

    state["tutor_adaptive_data"] = {
        "quiz_score": body.quiz_score.model_dump() if body.quiz_score else existing_data.get("quiz_score"),
        "focus_areas": new_areas,
    }
    await wf.asave_session(session=existing)
    logger.info(
        "Adaptive data persisted — session_id=%s focus_areas=%s",
        session_id, new_areas,
    )
    return {"status": "ok", "focus_areas": new_areas}
```

Do NOT add a `@limiter.limit(...)` decorator — this is a low-frequency internal endpoint. Do NOT add rate limiting unless explicitly required.
